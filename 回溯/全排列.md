#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=10;

vector<int> path;
bool flag[N];
void dfs(int n){
    if(path.size()==n){
        for(int i=0;i<path.size();i++) cout<<path[i]<<' ';
        cout<<endl;
        return ;
    }
    for(int i=1;i<=n;i++){
        if(!flag[i]) {
            path.push_back(i);
            flag[i]=true;
            dfs(n);
            path.pop_back();
            flag[i]=false;
        }
    }
}
int main(){
    int n;
    cin>>n;
    dfs(n);
    
    
    return 0;
}
